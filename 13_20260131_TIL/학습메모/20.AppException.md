    package com.aloha.teamproject.common.exception;

    import lombok.Getter;

    @Getter
    // @Getter = 롬복 어노테이션, 클래스내의 모든 필드에 대해서 getter 메서드 자동 생성
    // 즉, private 필드에 접근 할 수 있는 public 메서드를 자동으로 만들어줌
    public class AppExceptioncopy extends RuntimeException {
    // public class = 뜻 : 클래스선언
    // AppExceptioncopy : 클래스 이름
    // extends RuntimeException { : RuntimeException 클래스를 상속받음

    private final ErrorCode errorCode;
    // private = 접근제한자, final = 수정 불가능한, ErrorCode = 자료형, errorCode = 변수이름
    // 전체 해석: 이 클래스는 접근이 제한된 수정 불가능한 ErrorCode 타입의 errorCode 변수를 가짐
    // ErrorCode = int, String, boolean 과 같이 에러코드 정보를 담는 사용자 정의 자료형

    public AppExceptioncopy(ErrorCode errorCode) {
        // public: 접근제한자 설정
        // AppExeptionCopy: 접근제한자가 public인 생성자 메서드 이름
        //(ErrorCode errrorcode) : ErrorCode 타입의 errorCode 매개변수를 받아 생성자 메서드 실행

        super(errorCode.getMessage());
        // super() = 부모 클래스의 생성자 메서드를 호출
        // 부모는 누구? => RuntimeException
        // () = errorCode 객체의 getMessage() 뜻: 에러 메시지를 반환하는 메서드
        // 메서드를 호출하여 에러 메시지를 가져와 부모 생성자에 전달
        // 전체 해석: 부모 클래스(RuntimeException)의 생성자를 호출하여 에러 메시지를 생성
        this.errorCode = errorCode;
        // this. = 현재 클래스의 인스턴스를 가리킴
        // 인스턴스 = 클래스로부터 생성된 객체
        // this.errorCode = 현재 클래스의 errorCode 변수
        // = errorCode; = 매개변수로 전달된 errorCode 값을 할당
        // 자바에서의 = 뜻: 대입 연산자, 오른쪽 값을 왼쪽 변수에 할당
        // 전체해석: errorCode 매개변수값을 this.errorCode 변수에 할당        
    }
}
